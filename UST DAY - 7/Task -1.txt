Customer:

Attributes: customerId, name, email, password

Methods: login(), viewOrderHistory()

Attributes: bookId, title, author, price, ISBN, stockQuantity

Methods: getDetails(), updateStock()

Attributes: cartId, items (list of CartItems)

Methods: addBook(), removeBook(), updateQuantity(), calculateTotal()

CartItem:

Attributes: book: Book, quantity

Methods: calculateItemTotal()

Order:

Attributes: orderId, orderDate, totalAmount, status, 
customer: Customer, orderItems: List<OrderItem>

Methods: placeOrder(), cancelOrder(), viewOrder()

Methods: calculateSubtotal()

Methods: login(), addBook(), updateBook(), removeBook()
